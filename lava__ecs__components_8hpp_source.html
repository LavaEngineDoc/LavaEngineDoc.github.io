133 void config(LightComponent& light, TransformComponent& tr) {
134 enabled = light.enabled_;
135 type = (int)light.type_;
136 pos[0] = tr.pos_.x;
137 pos[1] = tr.pos_.y;
138 pos[2] = tr.pos_.z;
139
140 // Matriz de identidad
141 glm::mat4 rotationMatrix = glm::mat4(1.0f);
142
143 // Aplicar rotaciones en el orden Z, Y, X (o el orden que prefieras)
144 rotationMatrix = glm::rotate(rotationMatrix, tr.rot_.z, glm::vec3(0.0f, 0.0f, 1.0f)); // Rotaci�n en Z
145 rotationMatrix = glm::rotate(rotationMatrix, tr.rot_.y, glm::vec3(0.0f, 1.0f, 0.0f)); // Rotaci�n en Y
146 rotationMatrix = glm::rotate(rotationMatrix, tr.rot_.x, glm::vec3(1.0f, 0.0f, 0.0f)); // Rotaci�n en X
147
148 // Obtener el vector forward (tercera columna de la matriz, invertido si Z negativo es forward)
149 glm::vec3 forwardVector = glm::vec3(rotationMatrix[2]);
150 forwardVector = glm::normalize(forwardVector);
151
152 dir[0] = forwardVector.x;
153 dir[1] = forwardVector.y;
154 dir[2] = forwardVector.z;
155
156 diff_color[0] = light.base_color_.x;
157 diff_color[1] = light.base_color_.y;
158 diff_color[2] = light.base_color_.z;
159
160 spec_color[0] = light.spec_color_.x;
161 spec_color[1] = light.spec_color_.y;
162 spec_color[2] = light.spec_color_.z;
163 linear_att = light.linear_att_;
164 quad_att = light.quad_att_;
165 constant_att = light.constant_att_;
166 shininess = light.shininess_;
167 strength = light.strength_;
168 spot_dir[0] = light.spot_dir_.x;
169 spot_dir[1] = light.spot_dir_.y;
170 spot_dir[2] = light.spot_dir_.z;
171 cutoff = cosf(glm::radians(light.cutoff_));
